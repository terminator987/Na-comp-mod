monthly_country_pulse = {
	on_actions = {
		glorpui_on_country_value_sync
	}
}

on_game_start_after_lobby_human_country = {
	on_actions = {
		glorpui_on_human_country_value_sync_init
	}
}

on_game_load_after_lobby_human_country = {
	on_actions = {
		glorpui_on_human_country_value_sync_init
	}
}

glorpui_on_country_value_sync = {
	effect = {
		if = {
			limit = { is_human = yes }
			glorpui_sync_avg_stats = yes
		}
	}
}

glorpui_on_human_country_value_sync_init = {
	effect = {
		glorpui_sync_avg_stats = yes
	}
}
